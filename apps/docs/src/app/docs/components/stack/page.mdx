# Stack

A layout component for arranging children in a horizontal or vertical stack with configurable gap.

## Example

<Preview code={`
<div style={{ display: "flex", flexDirection: "column", gap: 24 }}>
  <Stack direction="horizontal" gap="md">
    <Button variant="primary">One</Button>
    <Button variant="secondary">Two</Button>
    <Button variant="ghost">Three</Button>
  </Stack>
  <Stack direction="vertical" gap="sm">
    <Input placeholder="First" />
    <Input placeholder="Second" />
    <Input placeholder="Third" />
  </Stack>
</div>
`} />

## Props

<PropsTable data={[
  { name: "direction", type: '"horizontal" | "vertical"', default: '"vertical"', description: "Stack direction." },
  { name: "gap", type: '"xs" | "sm" | "md" | "lg" | "xl"', default: '"md"', description: "Space between children." },
  { name: "className", type: "string", default: "—", description: "Additional CSS class names." },
  { name: "children", type: "ReactNode", default: "—", description: "Stack children." },
]} />

## Code

```tsx
import { Stack } from "@orbi/react";

<Stack direction="horizontal" gap="md">
  <Button>A</Button>
  <Button>B</Button>
</Stack>
```
