# Avatar

A circular avatar component with image support and fallback initials. Available in three sizes.

## Example

<LivePlayground code={`
<div style={{ display: "flex", gap: 16, alignItems: "center" }}>
  <Avatar size="sm" fallback="SM" />
  <Avatar size="md" fallback="MD" />
  <Avatar size="lg" fallback="LG" />
  <Avatar
    size="lg"
    src="https://i.pravatar.cc/80?img=12"
    alt="User avatar"
    fallback="U"
  />
</div>
`} />

## Props

<PropsTable data={[
  { name: "src", type: "string", default: "—", description: "Image URL for the avatar." },
  { name: "alt", type: "string", default: '""', description: "Alt text for the avatar image." },
  { name: "fallback", type: "string", default: "—", description: "Text shown when image is missing or fails to load." },
  { name: "size", type: '"sm" | "md" | "lg"', default: '"md"', description: "Size of the avatar." },
  { name: "className", type: "string", default: "—", description: "Additional CSS class names." },
]} />

## Code

```tsx
import { Avatar } from "@orbi/react";

<Avatar src="/user.jpg" alt="User" fallback="JD" size="md" />
```
